                                                                              # Boost电路的工作模态

                                                                                  Boost电路的工作模态分析。

                                                                              # Boost工作模态

                                                                              • Boost变换器原理电路如下图所示。
                                                                              1. 开关管导通期间（TonT_{on}Ton​）

                                                                              开关管导通，电感电流在输入电压的作用下线性上升，电感处于储能阶段。此时依靠电容给负载提供能量。

                                                                              电感两端电压：VL_in=Vin=L⋅didt=L⋅r⋅IinD⋅TsV_{L\_{in}} = V_{in} = L \cdot \frac{di}{dt} = L \cdot \frac{r \cdot I_{in}}{ D \cdot T_s}VL_in​=Vin​=L⋅dtdi​=L⋅D⋅Ts​r⋅Iin​​

                                                                              其中DDD为占空比，TsT_sTs​为开关周期，rrr为电感电流纹波率。

                                                                              1. 开关管断开期间1（ToffT_{off}Toff​）

                                                                              开关管断开由于电感电流不能突变，电感上的电压极性反向，电感和输入电源一起，通过二极管向输出电容和负载释放能量，所以Boost变换器能实现升压的功能。

                                                                              电感两端电压：VL_out=(Vin−Vout)=L⋅didt=L⋅IoToffV_{L\_{out}} = (V_{in} - V_{out})= L \cdot \frac{di}{dt} = L \cdot \frac{I_{o}}{T_{off}}VL_out​=(Vin​−Vout​)=L⋅dtdi​=L⋅Toff​Io​​

                                                                              其中DDD为占空比，TsT_sTs​为开关周期。

                                                                              1. 开关管断开期间2（ToffT_{off}Toff​）

                                                                              如果电感电流处于DCM条件下，此时依靠电容给负载提供能量。

                                                                              # 占空比D

                                                                              在CCM和BMC模式下，依据电感伏秒平衡特性：

                                                                              Vin⋅D⋅Ts+(Vin−Vout)⋅(1−D)⋅Ts=0 V_{in}\cdot D \cdot T_s + (V_{in} - V_{out})\cdot (1 - D )\cdot T_s = 0 Vin​⋅D⋅Ts​+(Vin​−Vout​)⋅(1−D)⋅Ts​=0

                                                                              Vin⋅D⋅Ts=(Vout−Vin)⋅(1−D)⋅Ts V_{in}\cdot D \cdot T_s = (V_{out} - V_{in})\cdot (1 - D )\cdot T_s Vin​⋅D⋅Ts​=(Vout​−Vin​)⋅(1−D)⋅Ts​

                                                                              Vin⋅D=(Vout−Vin)⋅(1−D) V_{in}\cdot D = (V_{out} - V_{in})\cdot (1-D) Vin​⋅D=(Vout​−Vin​)⋅(1−D)

                                                                              Vout⋅(1−D)=Vin⋅D+Vin⋅(1−D) V_{out}\cdot (1-D) = V_{in}\cdot D +V_{in}\cdot (1-D) Vout​⋅(1−D)=Vin​⋅D+Vin​⋅(1−D)

                                                                              Vout=Vin1−D V_{out} = \frac{V_{in}}{1-D} Vout​=1−DVin​​

                                                                              D=Vout−VinVout=1−VinVout D = \frac{V_{out} - V_{in}}{V_{out}} = 1-\frac{V_{in}}{V_{out}} D=Vout​Vout​−Vin​​=1−Vout​Vin​​

                                                                              # 电容计算

                                                                              开关管导通期间（TonT_{on}Ton​），负载电流IoutI_{out}Iout​由电容提供。

                                                                              Iout=C⋅dudt I_{out} = C \cdot \frac{du}{dt} Iout​=C⋅dtdu​

                                                                              ΔVout=IoutC⋅D⋅Ts \Delta V_{out} = \frac{I_{out}}{C} \cdot D \cdot T_s ΔVout​=CIout​​⋅D⋅Ts​

                                                                              其中，ΔVout\Delta V_{out}ΔVout​为电容电压纹波。

                                                                              C=IoutΔVout⋅D⋅Ts C = \frac{I_{out}}{\Delta V_{out}} \cdot D \cdot T_s C=ΔVout​Iout​​⋅D⋅Ts​

                                                                              # CCM与DCM的判定

                                                                              如果在一个开关周期结束前，电感电流仍然大于零，我们通常将这种工作状态称为电感电流连续模式（CCM）；如果在一个开关周期结束前某一个时段，电感电流就已经为零，我们通常将这种工作状态称为电感电流断续模式（DCM）。那么如何判断一个电感电流是工作在CCM还是DCM呢?

                                                                              在CCM和DCM之间，有一个临界工作状态，即在一个开关周期结束时，电感电流正好等于零，我们通常将这种工作状态称为电感电流临界连续模式（BCM）。

                                                                              对于BCM，其波形如下图所示。 从波形中可以看到，电感电流的平均值与电感电流之间的关系为：

                                                                              Iavg=ΔIL2 I_{avg} = \frac{\Delta I_L}{2} Iavg​=2ΔIL​​

                                                                              # 电感电流纹波率

                                                                              为了描述电感电流的变化程度，引入了电感电流纹波率这个概念。通常定义电流的纹波率指的是电流的交流成分与直流成分的比值:

                                                                              r=ΔIIL=2⋅IACIDC r = \frac{\Delta I}{I_L} = \frac{2 \cdot I_{AC}}{I_{DC}} r=IL​ΔI​=IDC​2⋅IAC​​

                                                                              IAC=ΔI2 I_{AC} = \frac{\Delta I}{2} IAC​=2ΔI​

                                                                              其中ILI_{L}IL​等于IDCI_{DC}IDC​，即平均电感电流。

                                                                              设计电感时，应首先确定电感电流的纹波率rrr，工程上rrr一般取值为0.3~0.5。

                                                                              # 电感计算

                                                                              • 方法一：按照电感电流纹波率rrr计算

                                                                              1. 开关管导通期间（TonT_{on}Ton​），电感两端电压：VL_in=Vin=L⋅didt=L⋅r⋅IinD⋅TsV_{L\_{in}} = V_{in} = L \cdot \frac{di}{dt} = L \cdot \frac{r \cdot I_{in}}{ D \cdot T_s}VL_in​=Vin​=L⋅dtdi​=L⋅D⋅Ts​r⋅Iin​​

                                                                              2. 根据能量守恒，有 Vout⋅Iout=Vin⋅IinV_{out} \cdot I_{out} = V_{in} \cdot I_{in}Vout​⋅Iout​=Vin​⋅Iin​，即 Iin=Vout⋅IoutVinI_{in} = \frac{V_{out} \cdot I_{out}}{V_{in}}Iin​=Vin​Vout​⋅Iout​​

                                                                              3. L=Vin⋅D⋅Tsr⋅Iin=Vin2⋅D⋅Tsr⋅Vout⋅IoutL = \frac{V_{in}\cdot D \cdot T_s}{r \cdot I_{in}} = \frac{V_{in}^2 \cdot D \cdot T_s}{r \cdot V_{out} \cdot I_{out} }L=r⋅Iin​Vin​⋅D⋅Ts​​=r⋅Vout​⋅Iout​Vin2​⋅D⋅Ts​​

                                                                              • 方法二：计算最小电感量

                                                                              1. 当电感电流需要工作在CCM模式下，则需要满足Iavg>ΔI2I_{avg} > \frac{\Delta I}{2}Iavg​>2ΔI​

                                                                              2. 根据能量守恒，有 Vout⋅Iout=Vin⋅IinV_{out} \cdot I_{out} = V_{in} \cdot I_{in}Vout​⋅Iout​=Vin​⋅Iin​。而且Vout=Vin1−DV_{out} = \frac{V_{in}}{1-D}Vout​=1−DVin​​

                                                                              3. Vout⋅Iout=Vout⋅(1−D)⋅IinV_{out} \cdot I_{out} = V_{out} \cdot (1-D) \cdot I_{in}Vout​⋅Iout​=Vout​⋅(1−D)⋅Iin​

                                                                              4. Iin=Iout1−D=IavgI_{in} = \frac{I_{out}}{1-D} = I_{avg}Iin​=1−DIout​​=Iavg​

                                                                              5. 开关管导通期间（TonT_{on}Ton​）,ΔI=VinL⋅D⋅Ts\Delta I = \frac{V_{in}}{L}\cdot D \cdot T_sΔI=LVin​​⋅D⋅Ts​

                                                                              6. ΔI2=12⋅VinL⋅D⋅Ts\frac{\Delta I}{2} = \frac{1}{2}\cdot \frac{V_{in}}{L}\cdot D \cdot T_s2ΔI​=21​⋅LVin​​⋅D⋅Ts​

                                                                              7. Iavg>ΔI2I_{avg} > \frac{\Delta I}{2}Iavg​>2ΔI​，即 Iout1−D>12⋅VinL⋅D⋅Ts\frac{I_{out}}{1-D} > \frac{1}{2}\cdot \frac{V_{in}}{L}\cdot D \cdot T_s1−DIout​​>21​⋅LVin​​⋅D⋅Ts​

                                                                              8. 1L<2⋅Iout(1−D)⋅Vin⋅D⋅Ts\frac{1}{L} < \frac{2\cdot I_{out}}{(1-D)\cdot V_{in}\cdot D \cdot T_s}L1​<(1−D)⋅Vin​⋅D⋅Ts​2⋅Iout​​

                                                                              9. L>Vin⋅D⋅Ts⋅(1−D)2⋅IoutL > \frac{V_{in}\cdot D \cdot T_s \cdot (1-D)}{2\cdot I_{out}}L>2⋅Iout​Vin​⋅D⋅Ts​⋅(1−D)​

                                                                              10. L>Vout⋅(1−D)⋅D⋅Ts⋅(1−D)2⋅IoutL > \frac{V_{out}\cdot (1-D)\cdot D \cdot T_s \cdot (1-D)}{2\cdot I_{out}}L>2⋅Iout​Vout​⋅(1−D)⋅D⋅Ts​⋅(1−D)​

                                                                              11. L>(1−D)2⋅D⋅Rload2⋅fsL > \frac{(1-D)^2 \cdot D \cdot R_{load}}{2\cdot f_s}L>2⋅fs​(1−D)2⋅D⋅Rload​​

                                                                              方法三：

                                                                              当电感电流连续，还可以根据电感电流是否始终大于负载电流分为和

                                                                              • ：当电感电流大于负载电流，电感给电容和负载供电。【电感放电（给负载供电）过程中，电容不参与放电（给负载供电）。】
                                                                              • ：当电感电流小于负载电流，电感和电容同时给输出负载供电。【电感放电（给负载供电）过程中，电容参与放电（电感电流小于负载电流[下图为10A的负载电流]，电容也开始给负载供电）。】
                                                                              和的电感临界值计算：

                                                                              1. 电感电流峰值： ILP=Iin+ΔI2I_{LP} = I_{in} + \frac{\Delta I}{2}ILP​=Iin​+2ΔI​

                                                                              2. 电感电流谷值：ILV=Iin−ΔI2=IoutI_{LV} = I_{in} - \frac{\Delta I}{2} = I_{out}ILV​=Iin​−2ΔI​=Iout​

                                                                              3. 根据前面的推导：Iin=Iout1−DI_{in} = \frac{I_{out}}{1-D}Iin​=1−DIout​​ 和ΔI2=12⋅VinL⋅D⋅Ts\frac{\Delta I}{2} = \frac{1}{2}\cdot \frac{V_{in}}{L}\cdot D \cdot T_s2ΔI​=21​⋅LVin​​⋅D⋅Ts​

                                                                              4. 步骤2可换为： Iout1−D−12⋅VinL⋅D⋅Ts=Iout\frac{I_{out}}{1-D} - \frac{1}{2}\cdot \frac{V_{in}}{L}\cdot D \cdot T_s= I_{out}1−DIout​​−21​⋅LVin​​⋅D⋅Ts​=Iout​

                                                                              5. 乘以IoutI_{out}Iout​可得，Iout1−D−12⋅Vin⋅IoutL⋅Iout⋅D⋅Ts=Iout\frac{I_{out}}{1-D} - \frac{1}{2}\cdot \frac{V_{in} \cdot I_{out}}{L \cdot I_{out}}\cdot D \cdot T_s= I_{out}1−DIout​​−21​⋅L⋅Iout​Vin​⋅Iout​​⋅D⋅Ts​=Iout​

                                                                              6. 替换VinV_{in}Vin​，可得，Iout1−D−12⋅Vout⋅(1−D)⋅IoutL⋅Iout⋅D⋅Ts=Iout\frac{I_{out}}{1-D} - \frac{1}{2}\cdot \frac{V_{out}\cdot (1-D) \cdot I_{out}}{L \cdot I_{out}}\cdot D \cdot T_s= I_{out}1−DIout​​−21​⋅L⋅Iout​Vout​⋅(1−D)⋅Iout​​⋅D⋅Ts​=Iout​

                                                                              7. 约掉IoutI_{out}Iout​，可得，11−D−1=12⋅Rload⋅(1−D)L⋅D⋅Ts\frac{1}{1-D} - 1 = \frac{1}{2}\cdot \frac{R_{load}\cdot (1-D) }{L }\cdot D \cdot T_s1−D1​−1=21​⋅LRload​⋅(1−D)​⋅D⋅Ts​

                                                                              8. D1−D=12⋅Rload⋅(1−D)L⋅D⋅Ts\frac{D}{1-D} = \frac{1}{2}\cdot \frac{R_{load}\cdot (1-D) }{L }\cdot D \cdot T_s1−DD​=21​⋅LRload​⋅(1−D)​⋅D⋅Ts​

                                                                              9. L=(1−D)2⋅Rload2⋅fsL = \frac{(1-D)^2 \cdot R_{load}}{2\cdot f_s}L=2⋅fs​(1−D)2⋅Rload​​

                                                                              • 使用Python计算电感量的代码如下。

import math

Vin = 100
Vout = 200
fs = 100e3
Ts = 1/fs
Pout = 2000
Rload = Vout**2/Pout
Iout = Vout/Rload

D = 1-Vin/Vout

# 电容计算
# 电容电压纹波0.01
C_r = 0.01
C = (Vout/Rload)*D*Ts/(C_r*Vout)      


# 电感计算方法一
# 电感电流纹波0.2
L_r = 0.2
L = (Vin*Vin*D*Ts)/(L_r*Vout*Iout)
 
# 电感计算方法二
L_BCM = (Rload*D*(1-D)*(1-D))/(2*fs)

# 电感计算方法三
L_K = (Rload*(1-D)*(1-D))/(2*fs)

print("Rload = ",format(Rload,'.6E'))
print("C = ",format(C,'.6E'))

print("\n")
print("L = ",format(L,'.6E'))
print("L_BCM = ",format(L_BCM,'.6E'))
print("L_K = ",format(L_K,'.6E'))

C =  2.500000E-05

                                                                                  其中，Gdo=2⋅VoD⋅M−12⋅M−1G_{do}=2\cdot \frac{V_o}{D}\cdot \frac{M-1}{2\cdot M-1}Gdo​=2⋅DVo​​⋅2⋅M−1M−1​

import matplotlib.pyplot as plt
import control as ctrl
import math


#推荐使用 control 0.8.3

s = ctrl.tf('s')

R = Vo**2/Pout      # 负载电阻
C = 2.500000E-05    # Boost的电容
L = 1.250000E-04    # Boost的电感
Dp = 1 - D          # 1- D
M = Vo/Vg           # 电压比

def transfer_function_current_CCM():
    # 传递函数
    Gid = ((2*Vo)/(Dp*Dp*R))*((1+s*R*C/2)/(1+((s*L+rL)/(Dp*Dp*R)) + (s*s*L*C+s*C*rL)/(Dp*Dp) ))
    # Bode 图绘制
    f = np.logspace(1, 6, 1000)
    w = 2 * np.pi * f
    mag,phase,omega=ctrl.bode_plot(Gid,w,dB=True,Hz=True,deg=True,Plot=True,label=r'$G_{id}$')


def transfer_function_voltage_DCM():
    Gdo = 2*(Vo/D)*((M-1)/(2*M-1))
    wp = (2*M-1)/((M-1)*R*C)
    # 传递函数
    Gvd_DCM = Gdo*(1/(1+s/wp))
    #Bode 图绘制
    f = np.logspace(1, 6, 1000)
    w = 2 * np.pi * f
    mag,phase,omega=ctrl.bode_plot(Gvd_DCM,w,dB=True,Hz=True,deg=True,Plot=True,label=r'$G_{vd\_DCM}$')


def transfer_function_voltage_CCM():
    Gdo = Vg/(Dp*Dp)
    wz1 = 1/(Rc*C)
    wz2 = (Dp*Dp*(R-rL))/(L)
    w0 = 1/np.sqrt(L*C)*np.sqrt((rL+Dp*Dp*R)/(R))
    Q = w0/(rL/L+1/(C*(R+Rc)))
    # 传递函数
    Gvd_CCM = Gdo*((1+s/wz1)*(1-s/wz2))/(1+s/(w0*Q)+s*s/(w0*w0))
    #Bode 图绘制
    f = np.logspace(1, 6, 1000)
    w = 2 * np.pi * f
    mag,phase,omega=ctrl.bode_plot(Gvd_CCM,w,dB=True,Hz=True,deg=True,Plot=True,label=r'$G_{vd\_CCM}$')


transfer_function_current_CCM()
transfer_function_voltage_CCM()
transfer_function_voltage_DCM()

plt.legend()
