                                                                        # PFC锁相环基本原理

                                                                        PFC锁相环基本原理。

                                                                        # 基本原理

                                                                        锁相环（Phase-Locked Loop ,PLL）是一个能够比较输出与输入相位差的反馈系统，利用外部输入的参考信号控制环路内部振荡信号的频率和相位，使震荡信号同步至参考信号。

                                                                        ​​​​

                                                                        PD：鉴相器（Phase Detector）

                                                                        LPF：低通滤波器（Low Pass Filter）

                                                                        VCO：压控振荡器（Voltage Controlled Oscillator）

                                                                        Vin：外部输入的参考信号

                                                                        Vout：输出的信号

                                                                        ‍

                                                                        由于开始时Vin和Vout存在时间差，Vin和Vout经过鉴相器，鉴相器检测到两个信号存在时间差（相位差）就会输出一个VPDV_{PD}VPD​的信号。VPDV_{PD}VPD​中存在高频和低频的成分，控制压控振荡器需要一个稳定的控制电压VcontV_{cont}Vcont​，所以VPDV_{PD}VPD​的输出需要增加一个滤波器，将VPDV_{PD}VPD​的高频成分滤除得到控制电压VcontV_{cont}Vcont​。相位差越大VcontV_{cont}Vcont​越大，VCO振荡得越快，Vout也会振荡得越快，从而消除Vin和Vout的相位差。通过反复的鉴相和调整，最终VCO的输出信号频率和输入信号频率一致，这时PLL电路就进入锁定状态。

                                                                        # 鉴相器

                                                                        ‍

                                                                        理想的鉴相器是相位差和输出电压成线性关系，相位差越大输出电压越大。

                                                                        # PFC锁相环

                                                                        锁相环目的为通过特定的计算方式构建与输入电网电压同相位的单位正弦量（峰峰值为1），在程序中首先建立一个表准正弦表，包含1024点。其中每个点之间的间隔（锁相环步长）对应查表的时间差。

                                                                        如何让被查表组合起来的图形最接近我们的输入正弦波，就是锁相环的目的。若正弦表查表时间为Ttable_searchT_{table\_search}Ttable_search​，（频率为Ftable_searchF_{table\_search}Ftable_search​实际为开关周期）电网周期时间为TlineT_{line}Tline​，则：

                                                                        Ntotal=Tline2⋅Ttable_search=Tline2⋅Ftable_search=Tline2⋅Fswitchn=Fswitch2⋅n⋅Fline N_{total}=\frac{T_{line}}{2 \cdot T_{table\_search}}=\frac{T_{line}}{2} \cdot F_{table\_search}=\frac{T_{line}}{2} \cdot \frac{F_{switch}}{n}=\frac{F_{switch}}{2 \cdot n \cdot F_{line}} Ntotal​=2⋅Ttable_search​Tline​​=2Tline​​⋅Ftable_search​=2Tline​​⋅nFswitch​​=2⋅n⋅Fline​Fswitch​​

                                                                        其中n为代表每隔n个开关周期查表一次，这里每个周期均查表，即为1，FswitchF_{switch}Fswitch​为开关频率。

                                                                        计算每次查表步长Step（间隔的点数）为：

                                                                        Step=1024Ntotal Step = \frac{1024}{N_{total}} Step=Ntotal​1024​

                                                                        以工频50Hz为例说明，若锁相环计算频率为45kHz（开关周期，也是计算周期），即在半个工频周期内，则在总共可以计算450次（45kHz/100Hz）。

                                                                        基于正弦数据表的长度为1024个点，则每次从表中取数step为2.276个间隔，转为Q12则等于PLLstep_dsp=PLLstep⋅Q12=9321PLL_{step\_dsp}= PLL_{step} \cdot Q12= 9321PLLstep_dsp​=PLLstep​⋅Q12=9321。

                                                                        当NtoatlN_{toatl}Ntoatl​值越大，锁相步长Step越小，锁相环越精确。要让锁相环的结果和实际波形没有误差，Step不是整数，而是小数。因此需要定标，以提高Step的精度，如下图所示：

                                                                        ​​

                                                                        锁相环函数具体程序流程图如图3.4所示。函数首先判断输入电压大于零还是小于零，根据输入电压极性进入不同的分支。

                                                                        当电压大于0，进入正半周期锁相处理。为消除误采样或电网畸变引起的过零误判断，需判断前半周期的计数值（大于100）。判断极性标志位是否为0（电压由负转正），若成立则置位极性标志位，锁相环输出量清零，计算上一周期的锁相环误差，若误差大于零，需要增加步长，若误差小于零，则减小步长，通过上面的计算，再经过若干周期处理后，会得到适合当前频率的步长。

                                                                        正半周期计数器累加，电压采样值累加，电路采样置累加。

                                                                        当电压小于0，进入负半周期锁相处理。同样的方式计算合理的锁相环步长。当在负半周期计数值达到100时，有效值计算标志位清置1。

                                                                        # PFC锁相环代码

                                                                        //锁相环结构体
struct _PLL
{
    char Polar;//Vac极性,电压负半边向正半边相互转变标志位
    long PosCnt;//正半周期输入电压计数@每个开关周期累计
    long NegCnt;//负半周期输入电压计数@每个开关周期累计
    long Step;//锁相环步长
    long StepSum;//锁相环输出的step叠加,实际上是锁相环位置量乘以Q12
    long CntErr;//锁相环误差
    long Cnt;//当前锁相环输出的正弦表位置
    long CntFianl;//最终当前锁相环输出的正弦表位置
};

extern struct _PLL PLL;
 
void VariableInit(void)
{
    PLL.Polar=0;//Vac极性,电压负半边向正半边相互转变标志位
    PLL.PosCnt=0;//正半周期输入电压计数@每个开关周期累计
    PLL.NegCnt=0;//负半周期输入电压计数@每个开关周期累计
    PLL.Step=9321;//锁相环步长Q12
    PLL.StepSum=0;//锁相环输出的step叠加,实际上是锁相环位置量乘以Q12
    PLL.CntErr=0;//锁相环误差
    PLL.Cnt=0;//当前锁相环输出的正弦表位置
    PLL.CntFianl=0;//最终当前锁相环输出的正弦表位置
}

/*
** ===================================================================
**     Funtion Name :  void PhaseLockLoop(void)
**     Description : 输入锁相环函数，求取合适的步长，输出正弦表的位置
**     Parameters  :
**     Returns     :
** ===================================================================
*/
//锁相环路表为1023表
#define MAX_SINE_CNT  1023
//锁相环路计算K值，实际调试出来的数据K
#define PLL_K  3
//有效值开始计算标志位，=1表明可以进行有效值计算
char RmsCalReadyFlag=0;
void PhaseLockLoop(void)
{
    //输入电压>0 ，输入相电压进入正半周期
    if(SADC.Vac  > 0)
    {
        //输入电压之前为负，由负转正，且前期负半周期技术值>100
        if((PLL.Polar == 0)&&(PLL.NegCnt>100))
        {
            //计数器清零
            PLL.PosCnt=0;
            PLL.NegCnt=0;
            //输入电压标志位为正
            PLL.Polar = 1;
            //PLL求和清掉
            PLL.StepSum= 0;
            //计算PLL误差
            PLL.CntErr = MAX_SINE_CNT  - PLL.Cnt;
            //闭环计算PLL的步长
            PLL.Step += PLL.CntErr * PLL_K;
            if(PLL.Step >13005)
                PLL.Step = 13005;
            else if(PLL.Step < 8389)
                PLL.Step = 8389;
        }
        //正半周计数器计数累加
        PLL.PosCnt ++;
        //电压,电流值累加Q15
        Vac.Sum += SADC.Vac;
        Iac.Sum += SADC.Imos;
    }
    else if(SADC.Vac< 0)//当输入电压<0 为输入相电压进入负半周期
     {
         //输入电压首次由正转负，且前期正半周期技术值>100
         if((PLL.Polar == 1)&&(PLL.PosCnt>100))
         {
             //计数器
             PLL.NegCnt=0;
             //输入电压首次由负转正标志位清零
             PLL.Polar = 0;
             //PLL求和清掉
             PLL.StepSum = 0;
             //计算PLL误差，
             PLL.CntErr= MAX_SINE_CNT  - PLL.Cnt;
             //闭环计算PLL的步长
             PLL.Step += PLL.CntErr* PLL_K;

             if(PLL.Step >13048)
                 PLL.Step = 13048;
             else if(PLL.Step < 8389)
                 PLL.Step = 8389;
         }
         //负半周计数器计数累加
         PLL.NegCnt++;
         //当进入负半周期后，在100次控制中断周期后，有效值计算标志位置1，开始计算有效值
         if(PLL.NegCnt == 100)
             //有效值计算标志位置1    ，可以计算有效值
             Flag.RmsCalReady = 1;
     }

    //锁相环累加 = 锁相环输出+步长
    PLL.StepSum+= PLL.Step;
    //最终锁相值
    PLL.Cnt=  PLL.StepSum >> 12;

    //超出1024数表时，返回从0开始
    if(PLL.Cnt> MAX_SINE_CNT)
       PLL.CntFianl = PLL.Cnt - MAX_SINE_CNT;
   else
       PLL.CntFianl  = PLL.Cnt;

}
