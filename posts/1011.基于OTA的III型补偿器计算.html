                                                                            # 基于OTA的III型补偿器计算

                                                                                基于OTA的III型补偿器计算。

                                                                            # 跨导型运算放大器(OTA)

                                                                            跨导型运算放大器（OTA）是一种压控电流源放大器，其输出电流与放大器的差分输入成正比。放大器跨导增益定义为gmg_mgm​。

                                                                            gm=ΔIoutΔVin g_m = \frac{\Delta I_{out}}{\Delta V_{in}} gm​=ΔVin​ΔIout​​

                                                                            上式中，输出电流变化量ΔIout\Delta I_{out}ΔIout​受控于OTA的输入电压变化量ΔVin\Delta V_{in}ΔVin​。所引起的放大器输出电流的变化量。

                                                                            在直流情况下，OTA输出电流等于其同相与反相输入电压差与gmg_mgm​的乘积，OTA两个输入端电压差为：

                                                                            ξ=V(+)−V(−) \xi = V_{(+)} - V_{(-)} ξ=V(+)​−V(−)​

                                                                            则输出电流为：

                                                                            Iout=ξ⋅gm I_{out} =\xi \cdot g_m Iout​=ξ⋅gm​

                                                                            OTA输出对输入的增益为：

                                                                            VoutVin=R1⋅gm \frac{V_{out}}{V_{in}} = R_1 \cdot g_m Vin​Vout​​=R1​⋅gm​

                                                                            从上式可以知道，我们可以通过不同组合的电阻、电容网络取代式中R1，这样就可以引入零点与极点了。

                                                                            G(s)=−(Rlower⋅gm)(Rlower+R1)R2⋅C1(C1+C2)1+1s⋅R2⋅C1(1+s⋅R2⋅C1⋅C2C1+C2)⋅s⋅C3⋅(R3+R1)+1[s⋅C3⋅(Rlower⋅R1Rlower+R1+R3)+1] G(s) = - \frac{(R_{lower} \cdot g_m)}{(R_{lower} + R_1)} \frac{R_2 \cdot C_1}{(C_1 + C_2)} \frac{1 + \frac{1}{s\cdot R_2 \cdot C_1}}{(1+s\cdot R_2 \cdot \frac{C_1 \cdot C_2}{C_1 + C_2})} \cdot \frac{s \cdot C_3 \cdot(R_3+R_1)+1}{[s\cdot C_3 \cdot(\frac{R_{lower} \cdot R_1}{R_{lower}+R_1}+R_3)+1]} G(s)=−(Rlower​+R1​)(Rlower​⋅gm​)​(C1​+C2​)R2​⋅C1​​(1+s⋅R2​⋅C1​+C2​C1​⋅C2​​)1+s⋅R2​⋅C1​1​​⋅[s⋅C3​⋅(Rlower​+R1​Rlower​⋅R1​​+R3​)+1]s⋅C3​⋅(R3​+R1​)+1​

                                                                            G(s)=−G0⋅(1+ωz1s)(1+sωp1)⋅(1+sωz2)(1+sωp2) G(s) = -G_0 \cdot \frac{(1+\frac{\omega_{z1}}{s})}{(1+\frac{s}{\omega_{p1}})} \cdot \frac{(1+\frac{s}{\omega_{z2}})}{(1+\frac{s}{\omega_{p2}})} G(s)=−G0​⋅(1+ωp1​s​)(1+sωz1​​)​⋅(1+ωp2​s​)(1+ωz2​s​)​

                                                                            其中，增益、零点和极点定义如下：

                                                                            G0=(Rlower⋅gm)(Rlower+R1)R2⋅C1(C1+C2) G_0 = \frac{(R_{lower} \cdot g_m)}{(R_{lower} + R_1)} \frac{R_2 \cdot C_1}{(C_1 + C_2)} G0​=(Rlower​+R1​)(Rlower​⋅gm​)​(C1​+C2​)R2​⋅C1​​

                                                                            ∣G(fc)∣=G0⋅1+(fz1fc)21+(fcfp1)2⋅1+(fcfz2)21+(fcfp2)2 |G(f_c)| = G_0 \cdot \frac{\sqrt{1+ (\frac{f_{z1}}{f_c})^2}}{\sqrt{1+ (\frac{f_c}{f_{p1}})^2}} \cdot \frac{\sqrt{1+ (\frac{f_c}{f_{z2}})^2}}{\sqrt{1+ (\frac{f_c}{f_{p2}})^2}} ∣G(fc​)∣=G0​⋅1+(fp1​fc​​)2​1+(fc​fz1​​)2​​⋅1+(fp2​fc​​)2​1+(fz2​fc​​)2​​

                                                                            然后我们接下来求R2R_2R2​，联立上面几式可得：

                                                                            R2=fp⋅G⋅(R1+Rlower)Rlower⋅gm⋅(fp1−fz1)⋅1+(fcfp1)21+(fz1fc)2⋅1+(fcfp2)21+(fcfz2)2 R_2 = \frac{f_p \cdot G \cdot(R_1 + R_{lower}) }{R_{lower} \cdot g_m \cdot(f_{p1} -f_{z1)}} \cdot \frac{\sqrt{1+ (\frac{f_c}{f_{p1}})^2}}{\sqrt{1+ (\frac{f_{z1}}{f_c})^2}} \cdot \frac{\sqrt{1+ (\frac{f_c}{f_{p2}})^2}}{\sqrt{1+ (\frac{f_c}{f_{z2}})^2}} R2​=Rlower​⋅gm​⋅(fp1​−fz1)​fp​⋅G⋅(R1​+Rlower​)​⋅1+(fc​fz1​​)2​1+(fp1​fc​​)2​​⋅1+(fz2​fc​​)2​1+(fp2​fc​​)2​​

                                                                            C1=12⋅π⋅fz1⋅R2 C_1 = \frac{1}{2\cdot \pi \cdot f_{z1} \cdot R_2} C1​=2⋅π⋅fz1​⋅R2​1​

                                                                            fp2fz2<R1Rlower+1 \frac{f_{p2}}{f_{z2}} < \frac{R_1}{R_{lower}}+1 fz2​fp2​​<Rlower​R1​​+1

                                                                            fp2fz2<VoutVref \frac{f_{p2}}{f_{z2}} < \frac{V_{out}}{V_{ref}} fz2​fp2​​<Vref​Vout​​

                                                                            从上式可以分析出，当输出电压比电压参考值大很多时，那么第二个极点和第二个零点之间的距离可以有比较大的选择，那么相位的提升会有较大的选择空间。但是如果输出电压与电压参考值接近时，比如5V输出，2.5V电压基准，则零极点之间的距离会很近。这样就大大约束了相位提升的空间。这个就是基于OTA的III型补偿器的局限性。

                                                                            根据《开关电源控制环路设计》的“6.4——3型补偿器：原点处极点与两个零极点对”章节，可知：

                                                                            极点与零点的比值rrr：

                                                                            r=fpfz r = \frac{f_p}{f_z} r=fz​fp​​

                                                                            相位提升量boost：

                                                                            boost=2⋅arctan⁡(r)−π2 boost =2\cdot \arctan(\sqrt{r}) - \frac{\pi}{2} boost=2⋅arctan(r​)−2π​

                                                                            # 案例

                                                                                假设需要设计一个19V输出，电压参考值为2.5V基准。假设分压上端电阻R1R_1R1​为66kohm，下端电阻RlowerR_{lower}Rlower​为10kohm，那么如何设计该参数呢？

                                                                            极点与零点之间的比值：

                                                                            fp2fz2<VoutVref=192.5=7.6 \frac{f_{p2}}{f_{z2}} < \frac{V_{out}}{V_{ref}}=\frac{19}{2.5}=7.6 fz2​fp2​​<Vref​Vout​​=2.519​=7.6

                                                                            第二对零极点可获得的最大相位提升量为：

                                                                            boost=(2⋅arctan⁡(7.6)−π2)⋅180°π=50° boost =(2\cdot \arctan(\sqrt{7.6}) - \frac{\pi}{2}) \cdot \frac{180\degree}{\pi}= 50\degree boost=(2⋅arctan(7.6​)−2π​)⋅π180°​=50°

                                                                            上述是针对III型补偿器的第二对零极点的约束导致的相位提升量，而III型补偿器中的第一对零极点并没有上述限制，所以其最大相位提升量为90°，则整个OTA的III型补偿器的相位最大提升量为：

                                                                            boostmax=90°+50°=140° boost_{max} = 90\degree + 50\degree = 140\degree boostmax​=90°+50°=140°

                                                                            所以我们在设计中选择穿越频率时，就必须意识到，在穿越频率处的相位提升量不能超过140°。

                                                                            我们不妨假设在穿越频率为1kHz位置，相位提升量为130°，且增益为15dB。那么我们首先需要把对数增益转化为十进制增益，即：

                                                                            G=10Gfc20=101520=5.6 G =10^{\frac{G_{f_c}}{20}} = 10^{\frac{15}{20}}=5.6 G=1020Gfc​​​=102015​=5.6

                                                                            我们先研究有限制条件的第二对零极点该如何设计：

                                                                            由前述可知，第二对零极点对的r值必须小于7.6，考虑到一定的裕量，这里选择r=7，那么第二对零极点就可以按照下面来放置：

                                                                            fz2=fcr=10007=378Hz f_{z2} = \frac{f_c}{\sqrt{r}}=\frac{1000}{\sqrt{7}}=378Hz fz2​=r​fc​​=7​1000​=378Hz

                                                                            fp2=fc⋅r=1000⋅7=2.6kHz f_{p2} = f_c \cdot \sqrt{r}=1000 \cdot \sqrt{7} = 2.6kHz fp2​=fc​⋅r​=1000⋅7​=2.6kHz

                                                                            我们再根据选定的第二对零极点的r值计算在穿越频率处，这对零极点的真实相位提升量为：

                                                                            boost2=(2⋅arctan⁡(7.0)−π2)⋅180°π≈48.6° boost_2 =(2\cdot \arctan(\sqrt{7.0}) - \frac{\pi}{2}) \cdot \frac{180\degree}{\pi} \approx 48.6\degree boost2​=(2⋅arctan(7.0​)−2π​)⋅π180°​≈48.6°

                                                                            那么如何设计第一对零极点呢？

                                                                            我们知道总相位提升量为130°，那么第一对零极点需要提升量为：

                                                                            boost1=130°−48.6°=81.4° boost_1 = 130\degree -48.6\degree = 81.4\degree boost1​=130°−48.6°=81.4°

import numpy as np
import matplotlib.pyplot as plt
import control as ctrl
import math

#推荐使用 control 0.8.3
s = ctrl.tf('s')

R1 = 66e3       #66k
Rlower = 10e3   #10k
gm = 100e-6     #us
fc = 1000.0     #Hz
G_fc_dB = 15.0  #dB
boost_all = 130 #130 degree
Vout = 19.0     #V
Vref = 2.5      #V

#########################################################
r2 = int(Vout/Vref)
# dec = (degree*pi)/180
# degree = dec*180/pi
boost2 = (2*np.arctan(np.sqrt(r2)) - np.pi/2)*180/np.pi
print("r2 =", r2)           
print("boost2 =", boost2) 

G = math.pow(10.0,(G_fc_dB/20))
print("G =", G) 

fz2 = fc/np.sqrt(r2)
fp2 = fc*np.sqrt(r2)
print("fz2 =", fz2, "Hz")
print("fp2 =", fp2, "Hz")

boost1 = boost_all - boost2
#boost1 = 80

r1 = (np.tan((boost1/2)/180*np.pi+(np.pi/4)))*(np.tan((boost1/2)/180*np.pi+(np.pi/4)))

print("boost1 =", boost1) 
print("r1 =", r1)

fz1 = fc/np.sqrt(r1)
fp1 = fc*np.sqrt(r1)

print("fz1 =", fz1, "Hz") 
print("fp1 =", fp1, "Hz") 



R2 = ((fp1*G*(R1+Rlower))/(Rlower*gm*(fp1 - fz1)))*(np.sqrt(1+(fc/fp1)*(fc/fp1)))*(np.sqrt(1+(fc/fp2)*(fc/fp2)))/(np.sqrt(1+ (fz1/fc)*(fz1/fc)))/(np.sqrt(1+(fc/fz2)*(fc/fz2)))
R3 = (R1*R1*fz2-R1*Rlower*(fp2-fz2))/((fp2-fz2)*(R1+Rlower))
C1 = 1/(2*np.pi*fz1*R2)

C2 = C1/(2*np.pi*C1*R2*fp1-1)
C3 = 1/(2*np.pi*(R1+R3)*fz2)


print("Rlower =", Rlower/math.pow(10.0,3), "k")   
print("R1 =", R1/math.pow(10.0,3), "k")   
print("R2 =", R2/math.pow(10.0,3), "k")   
print("R3 =", R3/math.pow(10.0,3), "k")   
print("C1 =", C1*math.pow(10.0,9), "nF")   
print("C2 =", C2*math.pow(10.0,12), "pF")   
print("C3 =", C3*math.pow(10.0,9), "nF")  


G0 = ((Rlower)/(Rlower+R1))*((R2*C1*gm)/(C1+C2))
wz1 = 1/(R2*C1)
wz2 = 1/((R1+R3)*C3)
wp1 = 1/(R2*(C1*C2)/(C1+C2))
wp2 = 1/((Rlower*R1/(Rlower+R1) +R3)*C3)
Hp = -G0*((1+wz1/s)*(1+s/wz2))/((1+s/wp1)*(1+s/wp2))

f = np.logspace(-1, 6, 1000)
w = 2 * np.pi * f
mag,phase,omega=ctrl.bode_plot(Hp,w,dB=True,Hz=True,deg=True,Plot=True,label='Hp')


ax1, ax2 = plt.gcf().axes  # get subplot axes

y1_limit = (ax1.get_ylim())
x1_limit = (ax1.get_xlim())
ax1.hlines(G_fc_dB, x1_limit[0], x1_limit[1], linestyles ="--", colors ="r")
ax1.vlines(fc, y1_limit[0], y1_limit[1], linestyles ="--", colors ="g")




y2_limit = (ax2.get_ylim())
x2_limit = (ax2.get_xlim())
ax2.hlines(boost_all+90, x2_limit[0], x2_limit[1], linestyles ="--", colors ="r")
ax2.vlines(fc, y2_limit[0], y2_limit[1], linestyles ="--", colors ="g")


plt.legend()
