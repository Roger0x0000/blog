传递函数推导

    借助Python可以方便推导出补偿器的传递函数：

import sympy as sp

#两电阻并联总值
def parallel(a,b):
    return (a*b)/(a+b)


R1,R2,C1,C2,s = sp.symbols('R1 R2 C1 C2 s')
 
Zf = parallel((R2 + 1/(C1*s)),(1/(C2*s)))
Zi = R1

H2 = sp.factor(Zf/Zi)
print("H2 = ",H2)  
#H2 =  (C1*R2*s + 1)/(R1*s*(C1*C2*R2*s + C1 + C2))

    假设需要为一个开关电源设计补偿电路，要求在穿越频率5kHz处需要提供Gfc=15dBG_{f_c}=15dBGfc​​=15dB，相位提升50°，该如何设计补偿电路？

  • 计算在穿越频率处提供幅值增益15dB的十进制增益：
G=101520=5.6234G = 10^{\frac{15}{20}}=5.6234G=102015​=5.6234

K=Tan[(Boost2)+45°]=Tan[(50°2)+45°]=Tan(70∙π180)=2.747477419 K = Tan[(\frac{Boost}{2}) + 45°] = Tan[(\frac{50°}{2}) + 45°]=Tan(\frac{70\bullet π}{180}) = 2.747477419 K=Tan[(2Boost​)+45°]=Tan[(250°​)+45°]=Tan(18070∙π​)=2.747477419

fp1=K∙fc=13.7373871kHzf_{p1}=K\bullet f_{c} = 13.7373871kHzfp1​=K∙fc​=13.7373871kHz

    fz1=fcK=1.819851171kHzf_{z1}=\frac{f_{c}}{K} = 1.819851171kHzfz1​=Kfc​​=1.819851171kHz

  • 按照K因子法计算C1、C2和R2

C2=12∗π∗fc∗G∗K∗R1=12∗π∗5000∗5.6234∗2.747477419∗10000=206pFC2 =\frac{1}{2*π*f_c*G*K*R1} = \frac{1}{2*π*5000*5.6234*2.747477419*10000} = 206pFC2=2∗π∗fc​∗G∗K∗R11​=2∗π∗5000∗5.6234∗2.747477419∗100001​=206pF

借助Python可以方便计算出补偿器的参数：


import numpy as np
import matplotlib.pyplot as plt
import control as ctrl
import math


#推荐使用 control 0.8.3

s = ctrl.tf('s')

R1 = 10e3 #10K
C1 = 1.3*math.pow(10.0,-9)  #1.3nF
R2 = 64.8e3 #64.8K
C2 = 206*math.pow(10.0,-12) #206pF

f_p0 = 1/(2*np.pi*R1*(C1+C2))           
f_z1 = 1/(2*np.pi*R2*C1)                
f_p1 = 1/(2*np.pi*R2*((C1*C2)/(C1+C2))) 

print("f_p0 =", f_p0, "Hz") 
print("f_z1 =", f_z1, "Hz") 
print("f_p1 =", f_p1, "Hz") 

#f_p0 = 10568.057310218814 Hz
#f_z1 = 1889.3036929237342 Hz
#f_p1 = 13812.093988073513 Hz

w_p0 = 2*np.pi*f_p0
w_z1 = 2*np.pi*f_z1
w_p1 = 2*np.pi*f_p1



H2 = -(w_p0/s)*(1+s/w_z1)/(1+s/w_p1)

f = np.logspace(-1, 6, 1000)
w = 2 * np.pi * f

mag,phase,omega=ctrl.bode_plot(H2,w,dB=True,Hz=True,deg=True,Plot=True,label='H2')

plt.legend()
plt.show()
