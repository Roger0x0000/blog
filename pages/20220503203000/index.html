STM32G0驱动-IWDG

本次实验目的是初始化STM32G071RB的独立看门狗。

原理图

按键线路

IWDG配置

看门狗的复位时间计算：

tIWDG=tLSI∗4∗2Prescaler∗(ReloadValue+1)t_{IWDG}=t_{LSI}*4*2^{Prescaler}*(ReloadValue+1)tIWDG​=tLSI​∗4∗2Prescaler∗(ReloadValue+1)
tIWDG=132KHz∗64∗(624+1)=31.25us∗64∗625=1250mst_{IWDG}=\frac{1}{32KHz}*64*(624+1)=31.25us*64*625=1250mstIWDG​=32KHz1​∗64∗(624+1)=31.25us∗64∗625=1250ms

关键程序

在main.c初始化中添加HAL_IWDG_Refresh()函数。


/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_IWDG_Init();
  /* USER CODE BEGIN 2 */
  //HAL_Delay(500);
  HAL_IWDG_Refresh(&hiwdg); 
  HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5,GPIO_PIN_RESET); //LED 
  HAL_Delay(500);

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
    //Key ON
    if(GPIO_PIN_RESET == HAL_GPIO_ReadPin(GPIOC,GPIO_PIN_13)) 
    {
        HAL_IWDG_Refresh(&hiwdg); 
        HAL_Delay(100);
        HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_5);
    }
    else
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5,GPIO_PIN_SET);
        HAL_Delay(2000);//IWDG will reset.
    }
  }
  /* USER CODE END 3 */
}
