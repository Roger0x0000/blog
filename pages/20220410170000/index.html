CRC-16-CCITT算法

  • The CRC-16-CCITT polynomial is 0x1021.
  • The initial register value must be 0xFFFF

    UInt16 calc_crc16_table(byte[] input, int length)
    {
        UInt16 crc = 0xFFFF; 

        for (int i = 0; i < length; i++)
        {
            UInt16 tableValue = crc16_ccitt_table[((crc >> 8) ^ input[i]) & 0xFF];

            crc = (UInt16)((crc << 8) ^ tableValue);
        }

        return crc;
    }

函数计算

    static UInt16 gen_crc16(byte[] u8Array, UInt16 u16Size)
    {
        UInt16 crcmask = 0xFFFF;
        UInt16 crcxor = 0;
        UInt16 polynom = 0x1021;
        UInt16 i, j, c, bit;
        UInt16 crc = 0x84CF, crchighbit = 0x8000;

        for (i = 0; i < u16Size; i++)
        {

            c = u8Array[i];
            for (j = 0x80; j!=0; j >>= 1)
            {

                bit = (UInt16)(crc & crchighbit);
                crc <<= 1;
                if ((c & j)!= 0)
                {
                    crc |= 1;
                }
                if (bit != 0)
                {
                    crc ^= polynom;
                }
            }
        }
        crc = (UInt16)(crc & 0xffff);
        for (i = 0; i < 16; i++)
        {

            bit = (UInt16)(crc & crchighbit);
            crc <<= 1;
            if (bit != 0)
            { 
                crc ^= polynom; 
            }
        }

        crc ^= crcxor;
        crc &= crcmask;
        return (crc);
    }
