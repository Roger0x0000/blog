Python代码实现AES-256-CTR



import os
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes

#AES-CTR-NoPadding

def pretty_print_hex(a, l=16, indent=''):
    """
    Format a list/bytes/bytearray object into a formatted ascii hex string
    """
    s = ''
    a = bytearray(a)
    for x in range(0, len(a), l):
        s += indent + ''.join(['%02X ' % y for y in a[x:x+l]]) + '\n'
    return s

key_256 =  bytearray([
        0xFF, 0x7A, 0x61, 0x7C, 0xE6, 0x91, 0x48, 0xE4,
        0xF1, 0x72, 0x6E, 0x2F, 0x43, 0x58, 0x1D, 0xE2,
        0xAA, 0x62, 0xD9, 0xF8, 0x05, 0x53, 0x2E, 0xDF,
        0xF1, 0xEE, 0xD6, 0x87, 0xFB, 0x54, 0x15, 0x3D
        ])

iv_128 = bytearray([
        0x00, 0x1C, 0xC5, 0xB7,
        0x51, 0xA5, 0x1D, 0x70, 0xA1, 0xC1, 0x11, 0x48,
        0x00, 0x00, 0x00, 0x01
        ])

data = bytearray([
		0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07,
		0x08, 0x09, 0x0A, 0x0B, 0x0C, 0x0D, 0x0E, 0x0F,
		0x10, 0x11, 0x12, 0x13, 0x14, 0x15, 0x16, 0x17,
		0x18, 0x19, 0x1A, 0x1B, 0x1C, 0x1D, 0x1E, 0x1F,
		0x20, 0x21, 0x22, 0x23
        ])



key = key_256 #os.urandom(32)
iv = iv_128 #os.urandom(16)

cipher = Cipher(algorithms.AES(key), modes.CTR(iv))
encryptor = cipher.encryptor()
encryptor_data = encryptor.update(data) + encryptor.finalize()
decryptor = cipher.decryptor()
decryptor_data = decryptor.update(encryptor_data) + decryptor.finalize()


print("Data: " + data.hex().upper())
print("Key:  " + key.hex().upper())
print("IV: "+ iv.hex().upper())

print("encryptor_data: " + encryptor_data.hex().upper())
print("decryptor_data: " + decryptor_data.hex().upper())

#print(pretty_print_hex(key,16))

print(pretty_print_hex(encryptor_data,16))
print(pretty_print_hex(key,16))
